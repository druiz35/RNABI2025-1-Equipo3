f_rosenbrock_2d <- function(x, y) {
  # Esta versin de la funcin se usa para graficar.
  z <- (100*((y-(x^2))^2))+((1-x)^2)
  return(z)
}

Definicin multidimencional

f_rosenbrock <- function(x){
  x_1 <- tail(x, -1)
  x <- head(x, -1)
  z <- sum((100*((x_1-(x^2))^2))+((1-x)^2))
  return(z)
}

f_rastrigin_2d <- function(x,y){
  # Esta versin de la funcin se usa para graficar.
  A <- 10
  z <- (A*2) + (x^2 - A*cos(2*pi*x)) + (y^2 - A*cos(2*pi*y))
  return(y)
}

Definicin multidimencional

f_rastrigin <-function(x){
  A <- 10
  n <- length(x)
  z <- (A*n) + sum(x^2 - A*cos(2*pi*x))
  return(z)
}

Definicin de Derivada Parcial

partial_dev <- function(x,i,fun,h=0.01){
    e <- x*0 # crea un vector de ceros de la misma longitud de x
    e[i] <- h
    y <- (fun(x+e)-fun(x-e))/(2*h)
  return(y)
}

Definicin del Gradiente

num_grad <- function(x,fun,h=0.01){
  # x: punto del espacio donde se debe evaluar el gradiente
  # fun: funcin para la que se desea calcular el gradiente en x
  # h: es el tamao de ventana para el clculo de la derivada numrica
  d <- length(x)
  y <- mapply(FUN=partial_dev,i=1:d,MoreArgs=list(x=x,h=h,fun=fun))
  return(y)
}

Definicin de Derivada del Gradiente

deriv_grad <- function(x,fun,i=1,h=0.01){
  # x: punto en el que se evala el gradiente
  # fun: funcin para la cual se calcula la derivada del gradiente respecto a la esima componente
  # i: i-sima componente del vector x con respecto a la que se deriva
    e <- x*0 # crea un vector de ceros de la misma longitud de x
    e[i] <- h
    y <- (num_grad(x+e,fun=fun,h=h)-num_grad(x-e,fun=fun,h=h))/(2*h)
    return(y)
}

Definicin de Matriz Hessiana

matriz_hessiana <- function(x,fun,h=0.01){
  # x: punto en el que se evala la matriz hessiana
  # fun: funcin a la que se le calcula la matriz hessiana en x
  # h: es el tamao de ventana para el clculo de la derivada numrica
  d <- length(x)
  y <- mapply(FUN=deriv_grad,i=1:d,MoreArgs=list(x=x,h=h,fun=fun),SIMPLIFY = TRUE)
  return(y)
}

Definicin de Optimizador Multivariado por Descenso del Gradiente

optimizador_mult_numdev <- function(x0,fun,max_eval=100,h=0.01,eta=0.01){
  x <- matrix(NA,ncol =length(x0), nrow = max_eval)
  x[1,] <- x0
  for (i in 2:max_eval){
    num_grad_fun <- num_grad(x[i-1,],fun,h)
    H <- matriz_hessiana(x[i-1,],fun,h)
    cambio <- - eta*solve(H)%*%num_grad_fun
    x[i,] <- x[i-1,] + cambio
    cambio_opt <- sqrt(sum((x[i-1,]-x[i,])^2))
    if (cambio_opt<0.00001){
      break
    }
  }
  return(x[1:i,])
}
